// packages/jobs/src/queues/bulkImportQueue.ts
import { Queue, Worker } from 'bullmq';
import { prisma } from '@formbricks/database';
import { logger } from '@formbricks/logger';
import csv from 'csv-parser';
import { ContactCreateInput, contactService } from '@formbricks/surveys';
import { Readable } from 'stream';
import fs from 'fs';
import { captureException } from '@sentry/node';

// Queue definition (already exists for other async tasks)
export const bulkImportQueue = new Queue('contact:bulkImport', {
  connection: redisConnection,
});

const BATCH_SIZE = 500;
const MAX_ROWS = 100_000; // optional hard limit

new Worker(
  'contact:bulkImport',
  async job => {
    const { orgId, filePath, userId } = job.data;
    const errors: { row: number; message: string }[] = [];
    const validRows: ContactCreateInput[] = [];

    const stream = fs.createReadStream(filePath);
    let rowNum = 0;

    await new Promise<void>((resolve, reject) => {
      stream
        .pipe(csv())
        .on('data', row => {
          rowNum++;
          // ---- validation -------------------------------------------------
          const { email, firstName, lastName, phone } = row;
          if (!email) {
            errors.push({ row: rowNum, message: 'Missing required field: email' });
            return;
          }
          // more domain‑specific validation (e.g., email format)
          validRows.push({ email, firstName, lastName, phone, organizationId: orgId });
          // ---- batch insert ------------------------------------------------
          if (validRows.length >= BATCH_SIZE) {
            stream.pause();
            insertBatch(validRows.splice(0, BATCH_SIZE))
              .then(() => stream.resume())
              .catch(err => reject(err));
          }
        })
        .on('end', async () => {
          // Insert any leftover rows
          if (validRows.length) {
            await insertBatch(validRows);
          }
          resolve();
        })
        .on('error', err => reject(err));
    });

    // --------------------------------------------------------------------
    // Helper to insert a batch atomically (use Prisma createMany)
    async function insertBatch(batch: ContactCreateInput[]) {
      try {
        await prisma.$transaction(async tx => {
          // `createMany` is fast and ignores duplicate‑key errors when `skipDuplicates:true`
          await tx.contact.createMany({
            data: batch,
            skipDuplicates: true,
          });
        });
      } catch (e) {
        // If a batch fails, capture row‑level errors via Prisma validation
        // For simplicity we abort the whole import and record the error.
        captureException(e);
        throw new Error(`Batch insert failed at row ${rowNum - batch.length + 1}: ${e}`);
      }
    }

    // --------------------------------------------------------------------
    // Store the import result (could be a separate table `contact_imports`)
    await prisma.contactImport.create({
      data: {
        organizationId: orgId,
        userId,
        totalRows: rowNum,
        successfulRows: rowNum - errors.length,
        failedRows: errors.length,
        errors: JSON.stringify(errors),
        fileName: path.basename(filePath),
      },
    });

    // Cleanup uploaded file
    fs.unlink(filePath, err => {
      if (err) logger.warn('Failed to delete temp CSV', { err, filePath });
    });

    logger.info('Bulk contact import completed', {
      orgId,
      totalRows: rowNum,
      successfulRows: rowNum - errors.length,
      failedRows: errors.length,
    });
  },
  {
    connection: redisConnection,
    concurrency: 2, // limited to avoid DB overload
  }
);
```

**Key points of the worker**

* **Streaming CSV** – `csv-parser` reads line‑by‑line, never loads the whole file.
* **Batch inserts** – `createMany` with `skipDuplicates:true` does bulk insert inside a short transaction; each batch is isolated, preventing long‑running table locks.
* **Per‑row validation** – errors are collected and persisted; the import does **not** abort on a single bad row – we simply skip it and continue (more user‑friendly). If you need an “all‑or‑none” policy, wrap the whole import in a single transaction and rollback on any error, but only after the batch size is proven safe (e.g., < 5 k rows).
* **Background processing** – the API returns instantly with a job id, satisfying serverless timeout limits.
* **Telemetry** – Sentry capture, structured logger, and a `contact_imports` table for audit.
* **Rate‑limit & auth** – matches existing contact endpoints.
